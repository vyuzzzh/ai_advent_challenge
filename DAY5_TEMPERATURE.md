```kotlin
private fun calculateSelfBLEU(responses: List<String>): Double {
    val wordSets = responses.map { response ->
        response.lowercase()
            .split(Regex("\\s+"))
            .filter { it.isNotBlank() }
            .toSet()
    }

    var totalSimilarity = 0.0
    var comparisons = 0

    for (i in wordSets.indices) {
        for (j in wordSets.indices) {
            if (i != j) {
                val intersection = wordSets[i].intersect(wordSets[j]).size
                val union = wordSets[i].union(wordSets[j]).size
                if (union > 0) {
                    totalSimilarity += intersection.toDouble() / union
                }
                comparisons++
            }
        }
    }

    return if (comparisons > 0) totalSimilarity / comparisons else 0.0
}
```

```kotlin
private fun calculateSemanticConsistency(responses: List<String>): Double {
    val wordFrequencies = responses.map { response ->
        response.lowercase()
            .split(Regex("\\s+"))
            .filter { it.isNotBlank() }
            .groupingBy { it }
            .eachCount()
    }

    val commonWords = wordFrequencies
        .flatMap { it.keys }
        .groupingBy { it }
        .eachCount()
        .filter { it.value == responses.size }
        .keys

    val totalWords = wordFrequencies
        .flatMap { it.keys }
        .toSet()
        .size

    return if (totalWords > 0) {
        commonWords.size.toDouble() / totalWords
    } else {
        0.0
    }
}
```

```kotlin
private fun calculateVariability(responses: List<String>): VariabilityMetrics {
    val lengths = responses.map { it.length.toDouble() }
    val avgLength = lengths.average()
    val lengthVariance = lengths.map { (it - avgLength).pow(2) }.average()
    val lengthStdDev = sqrt(lengthVariance)

    val uniqueWordsCounts = responses.map { response ->
        response.lowercase()
            .split(Regex("\\s+"))
            .filter { it.isNotBlank() }
            .toSet()
            .size
            .toDouble()
    }
    val avgUnique = uniqueWordsCounts.average()
    val uniqueVariance = uniqueWordsCounts.map { (it - avgUnique).pow(2) }.average()

    // –ù–æ—Ä–º–∞–ª–∏–∑–∞—Ü–∏—è –¥–ª—è —Å—Ç—Ä—É–∫—Ç—É—Ä–Ω–æ–≥–æ —Ä–∞–∑–Ω–æ–æ–±—Ä–∞–∑–∏—è
    val maxExpectedStdDev = avgLength * 0.3
    val normalizedStdDev = min(lengthStdDev / maxExpectedStdDev, 1.0)

    return VariabilityMetrics(
        lengthStdDev = lengthStdDev,
        uniqueWordsVariance = uniqueVariance,
        structuralDiversity = normalizedStdDev
    )
}
```
